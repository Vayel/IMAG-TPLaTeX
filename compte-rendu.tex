\begin{exo} On cherche à calculer la valeur de l'intégrale  
  \begin{equation*}
    I_{20} = \int_0^1 x^{20} e^{x} dx
  \end{equation*}
  \begin{enumerate}
  \item Donner une formule de récurrence pour calculer cette intégrale, puis évaluer $I_{20}$
    à partir de $I_0$
  \item Évaluer $I_{20}$ en utilisant le développement en série de $e^x$.
  \item Conclusions ?
  \end{enumerate}
\end{exo}

\begin{enumerate}
  \item 
    Notons $P_n(x) = x^n$. On a :

    \begin{align*}
      I_n &= \int_0^1 P_n(x) e^x dx \\
          &= [P_n \exp]_0^1 - \int_0^1 P_n' \exp \\
          &= [P_n \exp]_0^1 - \left[ [P_n' \exp]_0^1 - \int_0^1 P_n'' \exp \right] \\
          &= \left[ \sum_{k=0}^n (-1)^k P_n^{(k)} \exp \right]_0^1 \\
          &= \left[ \sum_{k=0}^n (-1)^k \frac{n!}{(n-k)!}x^{(n-k)} e^x \right]_0^1 \\
          &= \left[ (-1)^n n! e^x + \sum_{k=0}^{n-1} (-1)^k \frac{n!}{(n-k)!}x^{(n-k)} e^x \right]_0^1 \\
          &= (-1)^n n! (e - 1) + \sum_{k=0}^{n-1} (-1)^k \frac{n!}{(n-k)!} e
    \end{align*}

    Cette formule se montre par une récurrence simple :

    $I_0 = [e^x]_0^1 = e - 1 = (-1)^0 0! (e - 1)$.

    Si $n \in \N$ est tel que l'identité ci-dessus soit vérifiée, on a :

    \begin{align*}
      I_{n+1} &= \int_0^1 x^{n+1} e^x dx \\
              &= [x^{n+1} e^x]_0^1 - \int_0^1 (n+1) x^n e^x dx \\
              &= e - (n+1)I_n \\
    \end{align*}

    En évaluant $I_{20}$ avec Scilab, on obtient $- 129.26371$, ce qui est
    absurde vu que la fonction intégrée est positive. L'erreur provient des
    approximations sur les flottants.

  \item 
    On a :

    $e^x = \sum_{k=0}^{\infty} \frac {x^k} {k!}$.

    Donc :

    \begin{align*}
      I_n &= \int_0^1 \sum_{k=0}^{\infty} \frac {x^{k+n}}{k!} dx \\
          &= \sum_{k=0}^{\infty} \frac 1 {k!} \int_0^1 x^{k+n} dx (Beppo-Levi) \\
          &= \sum_{k=0}^{\infty} \frac 1 {k!} \frac 1 {k+n+1} \\
    \end{align*}

    Avec Scilab, on obtient $I_{20} = 0.1238038$.

  \item 
    Les deux résultats diffèrent : la première méthode est sujette aux 
    approximations dues aux flottants.
    
\end{enumerate}

\begin{exo} (théorique) Soit la fonction 1-périodique $f$, définie
  sur $[-\frac{1}{2}, \frac{1}{2}[$ par
  \begin{equation}
    \label{f} 
    f(x) = \left\{\begin{array}{lll} 
        -1 & \text{pour} & -\frac{1}{2} \le x \le 0 \,,\\ 
        1 & \text{pour} & 0< x < \frac{1}{2} \,.
      \end{array}\right.
  \end{equation} 
  Calculer la série de Fourier de $f$ et expliquer en
  quels points $x$ la formule ci-dessous est valide :
  \begin{equation}
    \label{Ff}
    f(x) = \frac{4}{\pi} \sum_{n=0}^{+\infty} \frac{\sin(2(2n+1)\pi x)}{2n+1} \,.
  \end{equation}
\end{exo}

  Soit $A$ une matrice carrée d'ordre $N$. Les valeurs propres de

  $D_k$, appelé disque de Gerschg\"orin est défini par:
  \begin{equation*}
    D_k = \{ z \in \C : |z - a_{kk}| \leq \Lambda_k =
    \sum\limits_{j=1,j\neq k}^N |a_{kj}| \} \,.
  \end{equation*}

    A = \left( \begin{array}{cccc}

                              &\leq \sum_{k \neq i} \abs {a_{ik}}
